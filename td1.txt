CREATE TABLE NIVEAU(
   NIV_NUM INT,
   NIV_PRIXUNITAIRECOPIE INT NOT NULL,
   PRIMARY KEY(NIV_NUM)
);

CREATE TABLE LANGUE(
   LAN_CODEL INT,
   LAN_LIBELLEL VARCHAR(50) NOT NULL,
   PRIMARY KEY(LAN_CODEL)
);

CREATE TABLE CORRECTEUR(
   COR_IDC INT,
   COR_NOMC VARCHAR(20) NOT NULL,
   COR_RUE VARCHAR(50) NOT NULL,
   COR_CP INT NOT NULL,
   COR_VILLE VARCHAR(20) NOT NULL,
   PRIMARY KEY(COR_IDC)
);

CREATE TABLE CLIENT(
   CLI_REF INT,
   CLI_NOM VARCHAR(20) NOT NULL,
   CLI_RUE VARCHAR(30) NOT NULL,
   CLI_CP INT NOT NULL,
   CLI_VILLE VARCHAR(30) NOT NULL,
   PRIMARY KEY(CLI_REF)
);

CREATE TABLE FACTURE(
   FACT_NUM INT,
   FAC_DATE DATE NOT NULL,
   CLI_REF INT NOT NULL,
   PRIMARY KEY(FACT_NUM),
   FOREIGN KEY(CLI_REF) REFERENCES CLIENT(CLI_REF)
);

CREATE TABLE REGLEMENT(
   REG_NUM INT,
   REG_DATE DATE NOT NULL,
   PRIMARY KEY(REG_NUM)
);

CREATE TABLE PACQUET(
   PAQ_NUMP INT,
   PAQ_NBCOPIES INT NOT NULL,
   PAQ_DATEREMISE DATE NOT NULL,
   PAQ_DATERETOUR DATE,
   FACT_NUM INT,
   COR_IDC INT NOT NULL,
   LAN_CODEL INT NOT NULL,
   NIV_NUM INT NOT NULL,
   PRIMARY KEY(PAQ_NUMP),
   FOREIGN KEY(FACT_NUM) REFERENCES FACTURE(FACT_NUM),
   FOREIGN KEY(COR_IDC) REFERENCES CORRECTEUR(COR_IDC),
   FOREIGN KEY(LAN_CODEL) REFERENCES LANGUE(LAN_CODEL),
   FOREIGN KEY(NIV_NUM) REFERENCES NIVEAU(NIV_NUM)
);

CREATE TABLE SPECIALISER(
   LAN_CODEL INT,
   COR_IDC INT,
   PRIMARY KEY(LAN_CODEL, COR_IDC),
   FOREIGN KEY(LAN_CODEL) REFERENCES LANGUE(LAN_CODEL),
   FOREIGN KEY(COR_IDC) REFERENCES CORRECTEUR(COR_IDC)
);

CREATE TABLE REGLER(
   FACT_NUM INT,
   REG_NUM INT,
   RGL_MONTANT INT NOT NULL,
   PRIMARY KEY(FACT_NUM, REG_NUM),
   FOREIGN KEY(FACT_NUM) REFERENCES FACTURE(FACT_NUM),
   FOREIGN KEY(REG_NUM) REFERENCES REGLEMENT(REG_NUM)
);
